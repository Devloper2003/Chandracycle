   227→function buildHormoneData(cycleLength: number) {
   228→  return Array.from({ length: cycleLength }, (_, i) => {
   229→    const day = i + 1
   230→    const ovulationDay = Math.max(14, cycleLength - 14)
   231→    const periodLen = 5
   232→    const follicularEnd = ovulationDay - 2
   233→    const ovulationEnd = ovulationDay + 1
   234→    const lutealMid = ovulationEnd + Math.floor((cycleLength - ovulationEnd) / 2)
   235→
   236→    // Estrogen curve
   237→    let estrogen: number
   238→    if (day <= periodLen) estrogen = 30 + day * 4
   239→    else if (day <= follicularEnd) estrogen = 50 + (day - periodLen) * 22
   240→    else if (day <= ovulationDay) estrogen = 240 + (day - follicularEnd) * 35
   241→    else if (day <= ovulationEnd) estrogen = 340 - (day - ovulationDay) * 80
   242→    else if (day <= lutealMid) estrogen = 180 + (day - ovulationEnd) * 10
   243→    else estrogen = Math.max(60, 240 - (day - lutealMid) * 22)
   244→
   245→    // Progesterone curve
   246→    let progesterone: number
   247→    if (day <= ovulationDay) progesterone = 5 + day * 0.5
   248→    else if (day <= lutealMid) progesterone = 12 + (day - ovulationDay) * 18
   249→    else progesterone = Math.max(8, 152 - (day - lutealMid) * 16)
   250→
   251→    return {
   252→      day,
   253→      estrogen: Math.round(estrogen),
   254→      progesterone: Math.round(progesterone),
   255→    }
   256→  })
   257→}

   259→function buildWeeklySymptomData(symptoms: SymptomEntry[]) {
   260→  const days = ['Mon', 'Tue', 'Wed', 'Thu', 'Fri', 'Sat', 'Sun']
   261→  // Build a map of last 7 days (Mon-Sun of current week)
   262→  const today = new Date()
   263→  const dayOfWeek = today.getDay() === 0 ? 6 : today.getDay() - 1 // 0=Mon
   264→  const monday = new Date(today)
   265→  monday.setDate(today.getDate() - dayOfWeek)
   266→
   267→  const buckets = days.map((name, idx) => {
   268→    const d = new Date(monday)
   269→    d.setDate(monday.getDate() + idx)
   270→    return {
   271→      name,
   272→      dateStr: d.toISOString().split('T')[0],
   273→      Cramps: 0,
   274→      Bloating: 0,
   275→      Headache: 0,
   276→      Fatigue: 0,
   277→      Mood: 0,
   278→    }
   279→  })
   280→
   281→  for (const s of symptoms) {
   282→    const bucket = buckets.find((b) => b.dateStr === s.date)
   283→    if (!bucket) continue
   284→    const sev = s.severity || 1
   285→    const cat = s.category.toLowerCase()
   286→    if (cat === 'cramps') bucket.Cramps += sev
   287→    else if (cat === 'bloating') bucket.Bloating += sev
   288→    else if (cat === 'headache') bucket.Headache += sev
   289→    else if (cat === 'fatigue') bucket.Fatigue += sev
   290→    else if (cat.includes('mood')) bucket.Mood += sev
   291→  }
   292→  return buckets.map(({ name, Cramps, Bloating, Headache, Fatigue, Mood }) => ({
   293→    name,
   294→    Cramps,
   295→    Bloating,
   296→    Headache,
   297→    Fatigue,
   298→    Mood,
   299→  }))
   300→}

   302→function buildRecentActivity(
   303→  cycles: CycleEntry[],
   304→  mood: MoodEntry[],
   305→  symptoms: SymptomEntry[],
   306→  sleep: SleepEntry[],
   307→) {
   308→  const items: Array<{
   309→    icon: typeof Droplets
   310→    title: string
   311→    time: string
   312→    color: string
   313→    ts: number
   314→  }> = []
   315→
   316→  const fmtTimeAgo = (iso: string) => {
   317→    const then = new Date(iso).getTime()
   318→    const now = Date.now()
   319→    const mins = Math.floor((now - then) / 60000)
   320→    if (mins < 1) return 'Just now'
   321→    if (mins < 60) return `${mins} min ago`
   322→    const hrs = Math.floor(mins / 60)
   323→    if (hrs < 24) return `${hrs} hour${hrs > 1 ? 's' : ''} ago`
   324→    const days = Math.floor(hrs / 24)
   325→    return `${days} day${days > 1 ? 's' : ''} ago`
   326→  }
   327→
   328→  // Most recent cycle
   329→  const latestCycle = cycles[0]
   330→  if (latestCycle) {
   331→    items.push({
   332→      icon: Droplets,
   333→      title: `Period logged — started ${latestCycle.startDate}`,
   334→      time: fmtTimeAgo(latestCycle.startDate),
   335→      color: 'text-rose-500',
   336→      ts: new Date(latestCycle.startDate).getTime(),
   337→    })

   365→      color: 'text-indigo-500',
   366→      ts: new Date(sl.date).getTime(),
