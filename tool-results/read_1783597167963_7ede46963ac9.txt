   105→type WorkoutCategory = 'strength' | 'cardio' | 'yoga' | 'pilates' | 'stretching'
   106→
   107→type Difficulty = 'Beginner' | 'Intermediate' | 'Advanced'
   108→
   109→interface Exercise {
   110→  name: string
   111→  duration: number // seconds for timed exercises
   112→  reps?: string // for rep-based exercises
   113→  description: string
   114→  tips: string
   115→  muscleGroup: string
   116→}
   117→
   118→interface Workout {
   119→  id: string
   120→  name: string
   121→  category: WorkoutCategory
   122→  duration: number // minutes
   123→  difficulty: Difficulty
   124→  calories: number
   125→  intensity: 'Low' | 'Medium' | 'Medium-High' | 'High'
   126→  muscleGroups: string[]
   127→  description: string
   128→  gradient: string
   129→  icon: React.ElementType
   130→  exercises: Exercise[]
   131→  phases: PhaseId[] // phases where this workout is recommended
   132→}
   133→

   146→  borderColor: string
   147→  textColor: string
   148→  badgeColor: string
   149→  icon: React.ElementType
   150→  routines: { name: string; duration: string; focus: string; exercises: string[] }[]
   151→  hormoneContext: string
   152→}

   966→interface PlayerState {
   967→  workout: Workout
   968→  exerciseIndex: number
   969→  remaining: number
   970→  isPlaying: boolean
   971→  isComplete: boolean
   972→  startedAt: number | null
   973→  completedExercises: number
   974→}
   975→
   976→function WorkoutPlayer({
   977→  workout,
   978→  onClose,
   979→}: {
   980→  workout: Workout
   981→  onClose: (completed: boolean, durationSec: number, calories: number) => void
   982→}) {
   983→  const [state, setState] = useState<PlayerState>({
   984→    workout,
   985→    exerciseIndex: 0,
   986→    remaining: workout.exercises[0]?.duration ?? 30,
   987→    isPlaying: true,
   988→    isComplete: false,
   989→    startedAt: Date.now(),
   990→    completedExercises: 0,
   991→  })
   992→
   993→  const totalExercises = workout.exercises.length
   994→  const currentExercise = workout.exercises[state.exerciseIndex]
   995→  const totalDuration = workout.exercises.reduce((sum, ex) => sum + ex.duration, 0)
   996→  const elapsedBeforeCurrent = workout.exercises
   997→    .slice(0, state.exerciseIndex)
   998→    .reduce((sum, ex) => sum + ex.duration, 0)
   999→  const overallProgress = ((elapsedBeforeCurrent + (currentExercise.duration - state.remaining)) / totalDuration) * 100
  1000→  const exerciseProgress = ((currentExercise.duration - state.remaining) / currentExercise.duration) * 100
  1001→
  1002→  // Working timer using useEffect + setInterval as required
  1003→  useEffect(() => {
  1004→    if (!state.isPlaying || state.isComplete) return
  1005→    const interval = setInterval(() => {
  1006→      setState((prev) => {
  1007→        if (prev.remaining > 1) {
  1008→          return { ...prev, remaining: prev.remaining - 1 }
  1009→        }
  1010→        // Exercise complete, advance
  1011→        const nextIndex = prev.exerciseIndex + 1
  1012→        if (nextIndex >= totalExercises) {
  1013→          return {
  1014→            ...prev,
  1015→            remaining: 0,
  1016→            isPlaying: false,
  1017→            isComplete: true,
  1018→            completedExercises: prev.completedExercises + 1,
  1019→          }
  1020→        }
  1021→        return {
  1022→          ...prev,
  1023→          exerciseIndex: nextIndex,
  1024→          remaining: prev.workout.exercises[nextIndex].duration,
  1025→          completedExercises: prev.completedExercises + 1,
  1026→        }
  1027→      })
  1028→    }, 1000)
  1029→    return () => clearInterval(interval)
  1030→  }, [state.isPlaying, state.isComplete, state.exerciseIndex, totalExercises])
  1031→
  1032→  const togglePlay = () => setState((p) => ({ ...p, isPlaying: !p.isPlaying }))
  1033→
  1034→  const goNext = useCallback(() => {
  1035→    setState((prev) => {
  1036→      const nextIndex = prev.exerciseIndex + 1
  1037→      if (nextIndex >= totalExercises) {
  1038→        return { ...prev, isComplete: true, isPlaying: false, remaining: 0 }
  1039→      }
  1040→      return {
  1041→        ...prev,
  1042→        exerciseIndex: nextIndex,
  1043→        remaining: prev.workout.exercises[nextIndex].duration,
  1044→        isPlaying: true,
  1045→      }
  1046→    })
  1047→  }, [totalExercises])
  1048→
  1049→  const goPrev = () => {
  1050→    setState((prev) => {
  1051→      if (prev.exerciseIndex === 0) return prev
  1052→      const prevIndex = prev.exerciseIndex - 1
  1053→      return {
  1054→        ...prev,
  1055→        exerciseIndex: prevIndex,
  1056→        remaining: prev.workout.exercises[prevIndex].duration,
  1057→        isPlaying: true,
  1058→      }
  1059→    })
  1060→  }
  1061→
  1062→  const elapsedSec = Math.floor((Date.now() - (state.startedAt ?? Date.now())) / 1000)
  1063→
  1064→  // Completion celebration screen
  1065→  if (state.isComplete) {
  1066→    const minutes = Math.floor(elapsedSec / 60)
  1067→    const seconds = elapsedSec % 60
  1068→    return (
  1069→      <div className="fixed inset-0 z-[100] flex items-center justify-center bg-gradient-to-br from-orange-500 via-rose-500 to-purple-600 p-4">
  1070→        <motion.div
  1071→          initial={{ scale: 0.8, opacity: 0 }}
  1072→          animate={{ scale: 1, opacity: 1 }}
  1073→          transition={{ type: 'spring', stiffness: 200, damping: 20 }}
  1074→          className="w-full max-w-md rounded-3xl bg-white/95 backdrop-blur-xl p-8 text-center shadow-2xl"
  1075→        >
  1076→          <motion.div
  1077→            initial={{ scale: 0, rotate: -180 }}
  1078→            animate={{ scale: 1, rotate: 0 }}
  1079→            transition={{ type: 'spring', stiffness: 200, damping: 15, delay: 0.2 }}
  1080→            className="mx-auto mb-6 flex h-24 w-24 items-center justify-center rounded-full bg-gradient-to-br from-orange-400 to-rose-500 shadow-lg"
  1081→          >
  1082→            <PartyPopper className="h-12 w-12 text-white" />
  1083→          </motion.div>
  1084→          <h2 className="text-3xl font-bold text-gray-900 mb-2">Workout Complete!</h2>
  1085→          <p className="text-gray-600 mb-6">Amazing work! You crushed {workout.name}.</p>
  1086→          <div className="grid grid-cols-3 gap-3 mb-8">
  1087→            <div className="rounded-2xl bg-orange-50 p-4">
  1088→              <Flame className="h-6 w-6 mx-auto text-orange-500 mb-1" />
  1089→              <div className="text-2xl font-bold text-gray-900">{workout.calories}</div>
  1090→              <div className="text-xs text-gray-500">Calories</div>
  1091→            </div>
  1092→            <div className="rounded-2xl bg-rose-50 p-4">
  1093→              <Clock className="h-6 w-6 mx-auto text-rose-500 mb-1" />
  1094→              <div className="text-2xl font-bold text-gray-900">{minutes}:{seconds.toString().padStart(2, '0')}</div>
  1095→              <div className="text-xs text-gray-500">Minutes</div>
  1096→            </div>
  1097→            <div className="rounded-2xl bg-purple-50 p-4">
  1098→              <CheckCircle2 className="h-6 w-6 mx-auto text-purple-500 mb-1" />
  1099→              <div className="text-2xl font-bold text-gray-900">{totalExercises}</div>
  1100→              <div className="text-xs text-gray-500">Exercises</div>
  1101→            </div>
  1102→          </div>
  1103→          <Button
  1104→            onClick={() => onClose(true, elapsedSec, workout.calories)}
  1105→            className="w-full h-12 text-base bg-gradient-to-r from-orange-500 to-rose-500 hover:from-orange-600 hover:to-rose-600 text-white shadow-lg"
  1106→          >
  1107→            Done
  1108→          </Button>
  1109→        </motion.div>
  1110→      </div>
  1111→    )
  1112→  }
  1113→
  1114→  return (
  1115→    <div className="fixed inset-0 z-[100] flex flex-col bg-gradient-to-br from-slate-900 via-orange-950 to-rose-950 text-white">
  1116→      {/* Header */}
  1117→      <div className="flex items-center justify-between px-6 py-4 border-b border-white/10">
  1118→        <div>
  1119→          <p className="text-xs uppercase tracking-wider text-orange-300/80">Now Playing</p>
  1120→          <h2 className="text-lg font-semibold">{workout.name}</h2>
  1121→        </div>
  1122→        <Button
  1123→          variant="ghost"
  1124→          size="icon"
  1125→          onClick={() => onClose(false, elapsedSec, 0)}
  1126→          className="text-white hover:bg-white/10 rounded-full"
  1127→        >
  1128→          <X className="h-5 w-5" />
  1129→        </Button>
  1130→      </div>
  1131→
  1132→      {/* Overall progress */}
  1133→      <div className="px-6 pt-3">
  1134→        <div className="flex justify-between text-xs text-white/60 mb-1.5">
  1135→          <span>Exercise {state.exerciseIndex + 1} of {totalExercises}</span>
  1136→          <span>{Math.round(overallProgress)}% complete</span>
  1137→        </div>
  1138→        <div className="h-1.5 w-full rounded-full bg-white/10 overflow-hidden">
  1139→          <motion.div
  1140→            className="h-full bg-gradient-to-r from-orange-400 to-rose-400"
  1141→            animate={{ width: `${overallProgress}%` }}
  1142→            transition={{ duration: 0.5 }}
  1143→          />
  1144→        </div>
  1145→      </div>
  1146→
  1147→      {/* Main content */}
  1148→      <div className="flex-1 flex flex-col items-center justify-center px-6 py-8 overflow-y-auto">
  1149→        <motion.div
  1150→          key={state.exerciseIndex}
  1151→          initial={{ opacity: 0, y: 20 }}
  1152→          animate={{ opacity: 1, y: 0 }}
  1153→          transition={{ duration: 0.4 }}
  1154→          className="text-center w-full max-w-lg"
  1155→        >
  1156→          <p className="text-sm uppercase tracking-wider text-orange-300/80 mb-2">
  1157→            {currentExercise.muscleGroup}
  1158→          </p>
  1159→          <h1 className="text-3xl md:text-4xl font-bold mb-6">{currentExercise.name}</h1>
  1160→
  1161→          {/* Timer ring */}
  1162→          <div className="relative mx-auto mb-8 h-56 w-56">
