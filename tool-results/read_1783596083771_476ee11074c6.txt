   546→// ─── Circular Gauge Component ───────────────────────────────────────────────
   547→
   548→function CircularGauge({ value, size = 180 }: { value: number; size?: number }) {
   549→  const strokeWidth = 14
   550→  const radius = (size - strokeWidth) / 2
   551→  const circumference = 2 * Math.PI * radius
   552→  const progress = (value / 100) * circumference
   553→  const center = size / 2
   554→
   555→  const getColorZone = (val: number) => {
   556→    if (val <= 30) return { color: '#22c55e', label: 'Low Risk', bg: 'bg-green-50 dark:bg-green-950/20' }
   557→    if (val <= 55) return { color: '#eab308', label: 'Moderate Risk', bg: 'bg-yellow-50 dark:bg-yellow-950/20' }
   558→    if (val <= 75) return { color: '#f97316', label: 'High Risk', bg: 'bg-orange-50 dark:bg-orange-950/20' }
   559→    return { color: '#ef4444', label: 'Very High Risk', bg: 'bg-red-50 dark:bg-red-950/20' }
   560→  }
   561→
   562→  const zone = getColorZone(value)
   563→
   564→  return (
   565→    <div className="flex flex-col items-center gap-3">
   566→      <div className={`relative rounded-2xl p-6 ${zone.bg}`}>
   567→        <svg width={size} height={size} className="transform -rotate-90">
   568→          {/* Background circle */}
   569→          <circle
   570→            cx={center}
   571→            cy={center}
   572→            r={radius}
   573→            fill="none"
   574→            stroke="currentColor"
   575→            strokeWidth={strokeWidth}
   576→            className="text-muted/30"
   577→          />
   578→          {/* Progress arc */}
   579→          <motion.circle
   580→            cx={center}
   581→            cy={center}
   582→            r={radius}
   583→            fill="none"
   584→            stroke={zone.color}
   585→            strokeWidth={strokeWidth}
   586→            strokeLinecap="round"
   587→            strokeDasharray={circumference}
   588→            initial={{ strokeDashoffset: circumference }}
   589→            animate={{ strokeDashoffset: circumference - progress }}
   590→            transition={{ duration: 1.5, ease: 'easeOut' }}
   591→          />
   592→        </svg>
   593→        <div className="absolute inset-0 flex flex-col items-center justify-center">
   594→          <motion.span
   595→            initial={{ opacity: 0, scale: 0.5 }}
   596→            animate={{ opacity: 1, scale: 1 }}
   597→            transition={{ delay: 0.5 }}
   598→            className="text-4xl font-bold"
   599→            style={{ color: zone.color }}
   600→          >
   601→            {value}
   602→          </motion.span>
   603→          <span className="text-[10px] text-muted-foreground font-medium uppercase tracking-wider mt-0.5">
   604→            / 100
   605→          </span>
   606→        </div>
   607→      </div>
   608→      <Badge
   609→        className="text-xs font-semibold px-3 py-1"
   610→        style={{ backgroundColor: zone.color + '20', color: zone.color, borderColor: zone.color + '40' }}
   611→        variant="outline"
   612→      >
   613→        {zone.label}
   614→      </Badge>
   615→    </div>
   616→  )
   617→}

   619→// ─── Progress Ring Component ────────────────────────────────────────────────
   620→
   621→function ProgressRing({ value, max, size = 80, color = '#ec4899' }: { value: number; max: number; size?: number; color?: string }) {
   622→  const strokeWidth = 6
   623→  const radius = (size - strokeWidth) / 2
   624→  const circumference = 2 * Math.PI * radius
   625→  const progress = (value / max) * circumference
   626→  const center = size / 2
   627→
   628→  return (
   629→    <div className="relative inline-flex items-center justify-center">
   630→      <svg width={size} height={size} className="transform -rotate-90">
   631→        <circle cx={center} cy={center} r={radius} fill="none" stroke="currentColor" strokeWidth={strokeWidth} className="text-muted/30" />
   632→        <motion.circle
   633→          cx={center}
   634→          cy={center}
   635→          r={radius}
   636→          fill="none"
   637→          stroke={color}
   638→          strokeWidth={strokeWidth}
   639→          strokeLinecap="round"
   640→          strokeDasharray={circumference}
   641→          initial={{ strokeDashoffset: circumference }}
   642→          animate={{ strokeDashoffset: circumference - progress }}
   643→          transition={{ duration: 1, ease: 'easeOut' }}
   644→        />
   645→      </svg>
   646→      <div className="absolute inset-0 flex items-center justify-center">
   647→        <span className="text-sm font-bold" style={{ color }}>{value}</span>
   648→      </div>
   649→    </div>
   650→  )
   651→}
