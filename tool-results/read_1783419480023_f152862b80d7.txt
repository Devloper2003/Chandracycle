   129тЖТ   129тЖТfunction debounce<T extends (...args: any[]) => void>(fn: T, delay: number) {
   130тЖТ   130тЖТ  let t: ReturnType<typeof setTimeout>
   131тЖТ   131тЖТ  return (...args: Parameters<T>) => {
   132тЖТ   132тЖТ    clearTimeout(t)
   133тЖТ   133тЖТ    t = setTimeout(() => fn(...args), delay)
   134тЖТ   134тЖТ  }
   135тЖТ   135тЖТ}
