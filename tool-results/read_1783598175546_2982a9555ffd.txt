   467→    setCart((prev) => {
   468→      const existing = prev.find((i) => i.product.id === product.id)
   469→      if (existing) {
   470→        return prev.map((i) =>
   471→          i.product.id === product.id ? { ...i, quantity: i.quantity + 1 } : i
   472→        )
   473→      }
   474→      return [...prev, { product, quantity: 1 }]
   475→    })
   476→    toast.success(`${product.name} added to cart`, {
   477→      description: formatINR(product.price),
   478→    })
   479→  }

   481→  const removeFromCart = (productId: string) => {
   482→    setCart((prev) => prev.filter((i) => i.product.id !== productId))
   483→  }
   484→

   486→    setCart((prev) =>
   487→      prev
   488→        .map((i) =>
   489→          i.product.id === productId
   490→            ? { ...i, quantity: Math.max(0, i.quantity + delta) }
   491→            : i
   492→        )
   493→        .filter((i) => i.quantity > 0)

   537→
   538→        <div className="relative flex flex-col lg:flex-row lg:items-center justify-between gap-6">
   539→          <div className="space-y-2 max-w-xl">
   540→            <div className="inline-flex items-center gap-2 px-3 py-1 rounded-full bg-white/15 backdrop-blur-sm text-xs font-medium">
   541→              <ShoppingBag className="h-3.5 w-3.5" />
   542→              ChandraCycle Marketplace
   543→            </div>
   544→            <h1 className="text-3xl sm:text-4xl lg:text-5xl font-bold tracking-tight">
   545→              <span className="bg-gradient-to-r from-white via-rose-50 to-amber-100 bg-clip-text text-transparent">
   546→                Wellness Market
   547→              </span>
   548→            </h1>
   549→            <p className="text-white/90 text-sm sm:text-base">
   550→              Curated products for your health journey
   551→            </p>
   552→          </div>
   553→
   554→          {/* Search + Cart */}
